#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    int m = -1;
    int end;
    cin >> n;
    vector<int> v(n + 1);
    vector<int> dp(n + 1, 0);
    for (int i = 1; i <= n; i++)
    {
        cin >> v[i];
        dp[i] = max(dp[i - 1] + v[i], v[i]);
        if (dp[i] > m)
        {
            m = dp[i];
            end = i;
        }
    }
    if (m < 0)
    {
        cout << 0 << ' ' << v[1] << ' ' << v[n];
        return 0;
    }
    int begin = end;
    int temp = m;
    for (begin = end; begin > 0; begin--)
    {
        temp -= v[begin];
        if (temp == 0)
        {
            break;
        }
    }
    cout << m << ' ' << v[begin] << ' ' << v[end];
    return 0;
}
