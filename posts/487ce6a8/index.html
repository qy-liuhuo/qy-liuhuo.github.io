#include <bits/stdc++.h>
using namespace std;
struct node
{
    int y;
    int r;
    int prenum;
    int p;
} s[100002];
bool cmp(node a, node b)
{
    return a.y > b.y;
}
int main()
{
    int n, sum0 = 0;
    cin >> n;
    for (int i = 0; i < n; i++)
    {
        cin >> s[i].y >> s[i].r;
        if (s[i].r == 0)
        {
            sum0++;
        }
    }
    sort(s, s + n, cmp);
    int pre1 = 0;
    for (int i = 0; i < n; i++)
    {
        s[i].prenum = pre1;
        int j, t = 0;
        s[i].p = pre1;
        for (j = i; j < n && s[j].y == s[i].y; j++)
        {

            s[i].p += s[j].r;
        }
        //cout << s[i].y << ' ' << s[i].r << ' ' << s[i].p << ' ' << sum0 - (j - s[i].p) << endl;
        s[i].p += sum0 - (j - s[i].p);
        pre1 += s[i].r;
    }
    int max = -1, result;
    for (int i = 0; i < n; i++)
    {
        //cout << s[i].y << ' ' << s[i].p << endl;
        if (max < s[i].p)
        {
            max = s[i].p;
            result = s[i].y;
        }
    }
    cout << result;
}
